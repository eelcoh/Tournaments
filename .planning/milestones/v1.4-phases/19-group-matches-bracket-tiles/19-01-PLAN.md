matchRowTile : Bool -> List (Element.Attribute msg) -> List (Element.Attribute msg)
matchRowTile isActive attrs =

        borderColor =
            if isActive then
                Color.orange
            else
                Color.terminalBorder

            if isActive then
                Color.white

                Color.grey
