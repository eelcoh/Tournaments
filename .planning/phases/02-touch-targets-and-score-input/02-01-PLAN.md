```elm
-- BEFORE
scoreButton_ : ButtonSemantics -> msg -> String -> Element msg
scoreButton_ semantics msg buttonText =
    let
        w =
            px 46
                |> width

        h =
            px 28
                |> height

        buttonLayout =
            UI.Style.scoreButton semantics [
                  w
                , h
                , onClick msg
                , centerX
                , centerY
                , height (px 26)
                ]
    in
    Element.row buttonLayout [ text buttonText ]

-- AFTER
scoreButton_ : ButtonSemantics -> msg -> String -> Element msg
scoreButton_ semantics msg buttonText =
    let
        w =
            px 46
                |> width

        buttonLayout =
            UI.Style.scoreButton semantics

                , height (px 44)
                , onClick msg
                , centerX
                , centerY
                ]
    in
    Element.row buttonLayout [ text buttonText ]
```
